```c
#include<iostream>
#include<string>
using namespace std;
int main()
{
	string s1, s2,ret;
	while (cin >> s1 >> s2)
	{
		while (s1.size()<s2.size())
			s1 = '0' + s1;
		while (s1.size()>s2.size())
			s2 = '0' + s2;
		//99 01
        ret.resize(s1.size());
		int add = 0, carry = 0;//标记每一位的和，是否进位
		for (int i = s1.size() - 1; i >= 0; i--)
		{
			add = (s1[i] - '0') + (s2[i] - '0') + carry;
			ret[i] =(add % 10) + '0';
			if (add / 10)
				carry = 1;
			else
				carry = 0;
		}
		if (carry)
			ret = '1' + ret;
		cout << ret << endl;
	}
	return 0;
}
```
